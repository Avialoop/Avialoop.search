function searchFlights(event) { event.preventDefault(); // Предотвращаем отправку формы const from = document.getElementById('from').value; const to = document.getElementById('to').value; const departure = document.getElementById('departure').value; const returnDate = document.getElementById('return').value; // Проверяем, что все поля заполнены if (!from || !to || !departure || !returnDate) { alert("Пожалуйста, заполните все поля."); return; } // Создаем URL с параметрами const searchUrl = `https://avialoop.github.io/search?from=${from}&to=${to}&departure=${departure}&return=${returnDate}`; window.location.href = searchUrl; // Перенаправляем на новый URL }
